utils.js

/**
 * @module utils
 */

/**
 * Adds two numbers
 * @param {Number} a 
 * @param {Number} b 
 * @returns {Number} sum
 */
export function sum (a, b) {
    return a + b
}

/**
 * adsf
 * @param {Number} a - number to subtract from
 * @param {Number} b - how much to subtract
 * @returns {Number}
 */
export function sub (a, b) {
    return a - b
}

/**
 * Times 
 * @param {Number} a - mult
 * @param {Number} b - mult
 * @returns {Number} - product
 */
 export function mult (a, b) {
    return a * b
}

/**
 * Times 
 * @param {Number} a - mult
 * @param {Number} b - mult
 * @returns {Number} - product
 */
 export function div (a, b) {
    return a / b
}
